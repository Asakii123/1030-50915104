Anime.js 動畫效果

Anime.js 官網

範例程式碼


            <div class="circle"></div>
            <script src="https://cdnjs.cloudflare.com/ajax/libs/animejs/3.2.1/anime.min.js"></script>
            <script>
                // 使用 Anime.js 建立移動和放大的動畫
                anime({
                    targets: '.circle',
                    keyframes: [
                        { translateX: 200, translateY: 0, scale: 1.5, duration: 1000 },
                        { translateX: 200, translateY: 200, scale: 2, duration: 1000 },
                        { translateX: 0, translateY: 200, scale: 1.5, duration: 1000 },
                        { translateX: 0, translateY: 0, scale: 1, duration: 1000 }
                    ],
                    loop: true,
                    easing: 'easeInOutQuad'
                });
                </script>
