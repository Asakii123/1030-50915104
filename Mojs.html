Mo.js 動畫效果

Mo.js 官網

範例程式碼


            <div id="effect-window"></div>
      
            <script>
            // 初始化 Mo.js 的爆炸效果
            const burst = new mojs.Burst({
            radius: { 20: 80 },           // 爆炸半徑範圍
            count: 10,                    // 爆炸的粒子數量
            children: {
                shape: 'circle',          // 粒子的形狀
                radius: { 5: 0 },         // 粒子半徑的變化
                scale: { 1: 0 },          // 粒子的縮放效果
                duration: 1000,           // 動畫持續時間
                easing: 'cubic.out',      // 動畫緩動效果
                stroke: '#ff6347',        // 粒子的顏色
                strokeWidth: { 2: 0 },    // 粒子的邊框寬度
            }
            });

            // 只在 effect-window 區域內觸發點擊事件
            document.getElementById('effect-window').addEventListener('click', function (e) {
            // 計算相對於 effect-window 的點擊位置
            const rect = e.currentTarget.getBoundingClientRect();
            const x = e.clientX - rect.left-650;  // 將座標轉換為相對位置
            const y = e.clientY - rect.top+430;

            // 設置爆炸效果的位置並播放動畫
            burst
                .tune({ x, y }) // 設定爆炸效果在視窗內的相對位置
                .replay();       // 播放動畫
            });
            </script>
